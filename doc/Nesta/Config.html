def diskcached_dir
  if settings.include?("diskcached_dir")
    unless settings.diskcached_dir =~ /^\//
      settings.diskcached_dir = File.join(Env.root, settings.diskcached_dir)
    end
    settings.diskcached_dir
  else
    File.join(Env.root, "diskcached")
  end
end
